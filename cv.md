#### [Link to kata](https://www.codewars.com/kata/array-deep-count) - 6 kyu
```
function deepCount(a){
  return  whileArrayOfArrays(a)
}

function whileArrayOfArrays(a){
  let n = 0
  let count = 0
  
  while (n<a.length){
    if (typeof(a[n])=='object'){
      count++
      count = count + whileArrayOfArrays(a[n])
    }  
    else {
      count++
    }
    n++
  }
  return count
}
```
